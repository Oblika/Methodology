
```bash
# Default Configuration
cat /etc/postfix/main.cf | grep -v "#" | sed -r "/^\s*$/d"

# To interact with the SMTP server, we can use the telnet
telnet <IP> <PORT>

# Snmpwalk is used to query the OIDs
snmpwalk -v2c -c public 10.129.14.128

# OneSixtyOne
sudo apt install onesixtyone
onesixtyone -c /opt/useful/seclists/Discovery/SNMP/snmp.txt 10.129.14.128

# Braa
sudo apt install braa
# Syntax
braa <community string>@<IP>:.1.3.6.*

braa public@10.129.14.128:.1.3.6.*

# Telnet - HELO/EHLO, nxjerin info nga serveri
HELO mail1.inlanefreight.htb
EHLO mail1
# Send email from SMTP
EHLO inlanefreight.htb

# Telnet - VRFY, enum perdoruesit ne sistem, ne varsi te konfigurimit
VRFY <name>

# Dangerous Settings, can send fake email
mynetworks = 0.0.0.0/0
```
