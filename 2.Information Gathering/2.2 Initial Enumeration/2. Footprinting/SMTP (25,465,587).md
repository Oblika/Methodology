# Default Configuration
cat /etc/postfix/main.cf | grep -v "#" | sed -r "/^\s*$/d"

# DNS lookup for mail servers for the specified domain
host -t MX microsoft.co

# DNS lookup for mail servers for the specified domain
dig mx inlanefreight.com | grep "MX" | grep -v ";"

# DNS lookup of the IPv4 address for the specified subdomain.
host -t A mail1.inlanefreight.htb.


# To interact with the SMTP server, we can use the telnet
telnet <IP> <PORT>

# Telnet - HELO/EHLO, nxjerin info nga serveri
HELO mail1.inlanefreight.htb
EHLO mail1

# Send email from SMTP
EHLO inlanefreight.htb

# SMTP VRFY Command:
- VRFY root
- VRFY new-user
- VRFY new-user

# EXPN Command
- EXPN john
- EXPN support-team

# RCPT TO Command
- MAIL <FROM:test@htb.com>
- RCPT TO:julio

# USER Command
- USER julio

# Dangerous Settings, can send fake email
mynetworks = 0.0.0.0/0
