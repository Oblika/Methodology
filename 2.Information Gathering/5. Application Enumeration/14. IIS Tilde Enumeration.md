```bash
# IIS Tilde eshte nje teknik per te zbuluar file, direktroit e fshehura ne disa :
# versions of Microsoft Internet Information Services (IIS) web servers.

# Tilde Enumeration using IIS ShortName Scanner
<https://github.com/irsdl/IIS-ShortName-Scanner>

#  need to install Oracle Java for Enumeration
<https://ubuntuhandbook.org/index.php/2022/03/install-jdk-18-ubuntu/>

# Syntax
java -jar iis_shortname_scanner.jar 0 5 <http://10.129.204.231/>

# Generate Wordlist
egrep -r ^transf /usr/share/wordlists/* | sed 's/^[^:]*://' > /tmp/list.txt

# Gobuster Enumeration
gobuster dir -u <http://10.129.204.231/> -w /tmp/list.txt -x .aspx,.asp
```
