```bash
# Log in page
<http://blog.inlanefreight.local/wp-login.php>

# Nje faqe WordPress dallohet nga rezultati qe jep /robots.txt, si wp-admin
curl -sS <http://blog.inlanefreight.local/robots.txt>

# header (për të parë redirect / server)
curl -I <http://blog.inlanefreight.local>

# Mund te shikojm versionin e WordPRess
curl -s <http://blog.inlanefreight.local> | grep WordPress
 
# Mund te shikojm info rreth "themes" qe permban WordPresi
curl -s <http://blog.inlanefreight.local/> | grep themes

# Shikojm Pluginet qe jan te instaluara
curl -s <http://blog.inlanefreight.local/> | grep plugins

# kërkim i shpejtë  për targa si "WordPress", "theme", "plugin"
curl -s <http://blog.inlanefreight.local> | egrep -i "wordpress|wp-content|wp-includes|theme|plugins"

# ---------- WPScan -------------

# Përditëso DB lokale të WPScan para çdo skanimi
wpscan --update

# Skanim i thjeshtë
wpscan --url <http://blog.inlanefreight.local> --output wpscan_basic.txt

# Enumerim i përdoruesve
sudo wpscan --url <http://blog.inlanefreight.local> --enumerate u

# Enumerim i plugin-eve popullore dhe të gjithë plugin-eve (ose vetëm vulnerable)
wpscan --url <http://blog.inlanefreight.local> --enumerate p,ap,vp

# Enumerim i themes (popular / all / vulnerable)
wpscan --url <http://blog.inlanefreight.local> --enumerate t,at,vt

# Të gjitha opsionet e rëndësishme të enumeration (shembull)
wpscan --url <http://blog.inlanefreight.local> --enumerate u,ap,at,tt,cb,dbe

# wpscan
# Nëse ke API token për WPVulnDB:
wpscan --url <http://blog.inlanefreight.local> --enumerate vp --api-token YOUR_TOKEN --output wpscan_vuln_plugins.json --format json
```
